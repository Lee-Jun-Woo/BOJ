const calculator = {
    add: function(a, b) {
        return a + b;
    },
    sub: function(a, b) {
        return a - b;
    },
    mul: function(a, b) {
        return a * b;
    },
    div: function(a, b) {
        return a / b;
    },
    pow: function(a, b) {
        return a ** b;
    },
};

console.log(calculator.add(2, 3));
const minus = calculator.sub(4, 3);
console.log(minus);
